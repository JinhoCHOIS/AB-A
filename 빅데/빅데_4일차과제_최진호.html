def remove_outlier_std(df):
    for i in range(0, len(df.iloc[1])):
        df = df[(np.abs(df.iloc[:,i] - df.iloc[:,i].mean()) < (3.6*df.iloc[:,i].std()))]
    return df
#3.6sigma를 벗어나는 data제거 함수
